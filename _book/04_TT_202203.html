df <- mtcars %>%
  make_long(cyl, vs, am, gear, carb)
ggplot(df, aes(x = x,
               node = node,
               next_x = next_x,
               next_node = next_node,
               fill = factor(node),
               label = node)) +
  geom_sankey() +
  geom_sankey_label() 
df1 <- mtcars %>%
  select(cyl, vs, am, gear, carb) %>%
  pivot_longer(everything()) %>%
  mutate(next_x = lead(.data$name),
         next_node = lead(.data$value)
  )
